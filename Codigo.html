Cálculo médio de download em Python 3


mb = float(input("Digite o tamanho do arquivo, em Gb: "))
vel = float(input("Digite a velocidade da internet, em mb/s: "))
tempo = (mb * 1000) // vel
horas = tempo // 3600
resto = tempo % 3600
minutos = resto // 60
segundos = resto % 60

print(horas, "horas,", minutos, "minutos e", segundos, "segundos")
